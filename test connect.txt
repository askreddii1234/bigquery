import pandas as pd
import datetime
import uuid

# Defining the SequenceGenerator class for generating sequences
class SequenceGenerator:
    def __init__(self, start=1):
        self.num = start
    
    def next_num(self):
        current = self.num
        self.num += 1
        return current

# Sample data for party table with dob and client columns
party_data = {
    'dob': [None, '1992-02-20', None, '1985-10-15'],
    'client': [str(uuid.uuid4()) for _ in range(4)]
}
df = pd.DataFrame(party_data)
df['result'] = df['dob'].apply(lambda x: 'ind' if x is not None else 'org')

# Adding party_id using the sequence generator
seq_gen_party_id = SequenceGenerator()
df['party_id'] = [seq_gen_party_id.next_num() for _ in range(len(df))]

# Extracting party_ids for party_internal_reference table
party_ids = df['party_id'].tolist()

# Create a new sequence generator for reference_id
seq_gen_reference_id = SequenceGenerator()

# Get the current system date
current_date = datetime.datetime.now().strftime('%Y-%m-%d')

# Map party_id to client ID
client_ids = df.set_index('party_id')['client'].to_dict()

    'id': [1, 2],
    'db_name': ['DB1', 'DB2'],
    'source_system_name': ['Database One', 'Database Two']

    'db_name': ['DB1' for _ in range(len(df))],

def save_dataframe(df, bucket_name, destination_blob_name, folder_name=None, is_debug=False):
    """Uploads a dataframe to GCS or saves locally based on debug flag."""
    # Convert dataframe to CSV
    output = StringIO()
    df.to_csv(output, index=False)
    output.seek(0)
    
    # If debug mode, save locally
    if is_debug:
        local_path = os.path.join(folder_name if folder_name else "", destination_blob_name)
        with open(local_path, 'w') as local_file:
            df.to_csv(local_file, index=False)
        return

    # If not in debug mode, save to GCS
    if folder_name:
        destination_blob_name = f"{folder_name}/{destination_blob_name}"

    storage_client = storage.Client()
    bucket = storage_client.bucket(bucket_name)
    blob = bucket.blob(destination_blob_name)
    blob.upload_from_file(output, content_type='text/csv')

import pandas as pd
import datetime
import uuid
import json
from google.cloud import bigquery

# Define the Google Cloud project and dataset ID
project_id = "your-project-id"
dataset_id = "your-dataset-id"

# Defining the SequenceGenerator class for generating sequences
class SequenceGenerator:
    def __init__(self, start=1):
        self.num = start
    
    def next_num(self):
        current = self.num
        self.num += 1
        return current

# Sample data for party table with dob, client, and party_type_code columns
party_data = {
    'dob': [None, '1992-02-20', None, '1985-10-15'],
    'client': [str(uuid.uuid4()) for _ in range(4)],
    'party_type_code': ['IND', 'ORG', 'IND', 'ORG']
}
party_df = pd.DataFrame(party_data)  # Create the 'party' dataframe

# Adding party_id using the sequence generator
seq_gen_party_id = SequenceGenerator()
party_df['party_id'] = [seq_gen_party_id.next_num() for _ in range(len(party_df))]

# Check if 'dob' is a valid date and apply 'ind' or 'org' accordingly
party_df['result'] = party_df['dob'].apply(lambda x: 'ind' if pd.notna(x) and pd.to_datetime(x, errors='coerce') is not pd.NaT else 'org')

# Print the 'party' table
print("Party Table:")
print(party_df)

# Save the 'party_df' DataFrame to a JSON file
json_file_path = 'party_data.json'
party_df.to_json(json_file_path, orient='records', lines=True)

print(f"Data saved to {json_file_path}")

# Load the data into BigQuery
client = bigquery.Client(project=project_id)
dataset_ref = client.dataset(dataset_id)

# Define the table name
table_name = "party"

# Create a job configuration
job_config = bigquery.LoadJobConfig(
    schema=[
        bigquery.SchemaField("party_id", "INTEGER"),
        bigquery.SchemaField("dob", "DATE"),
        bigquery.SchemaField("client", "STRING"),
        bigquery.SchemaField("party_type_code", "STRING"),
        bigquery.SchemaField("result", "STRING")
    ],
    source_format=bigquery.SourceFormat.NEWLINE_DELIMITED_JSON,
)

# Load the data from the JSON file into BigQuery
with open(json_file_path, "rb") as source_file:
    job = client.load_table_from_file(source_file, dataset_ref.table(table_name), job_config=job_config)

job.result()  # Wait for the job to complete

print(f"Data loaded into BigQuery table {project_id}.{dataset_id}.{table_name}")
